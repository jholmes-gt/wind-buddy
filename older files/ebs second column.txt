document.addEventListener('DOMContentLoaded', () => {
  const panelRoot = document.getElementById('ebsPanelRoot');
  const toggleBtn = document.getElementById('btn_endbringer');

  // Function to create and build the panel
  function createPanel() {
    const wrapper = document.createElement('div');
    wrapper.classList.add('panel-wrapper', 'hidden'); // Initially hidden

    // Create header
    const header = document.createElement('header');
    header.classList.add('panel-header');
    const headerText = document.createElement('h1');
    headerText.textContent = "Your Panel Header";
    header.appendChild(headerText);
    wrapper.appendChild(header);

    // Create the two columns container
    const container = document.createElement('div');
    container.classList.add('panel-container');

    // Create the first column
    const column1 = document.createElement('div');
    column1.classList.add('panel-column');
    column1.innerHTML = '<h2>Column One</h2><p>Content for the first column.</p>';

    // Create the second column
    const column2 = document.createElement('div');
    column2.classList.add('panel-column');
    column2.innerHTML = '<h2>Column Two</h2><p>Content for the second column.</p>';

    // Append columns to their container
    container.appendChild(column1);
    container.appendChild(column2);
    wrapper.appendChild(container);

    return wrapper;
  }

  // Create the panel and append it to the root element
  const panel = createPanel();
  panelRoot.appendChild(panel);

  // Add event listener to the button
  toggleBtn.addEventListener('click', () => {
    if (panel.classList.contains('hidden')) {
      panel.classList.remove('hidden');
      toggleBtn.textContent = 'Hide Panel';
    } else {
      panel.classList.add('hidden');
      toggleBtn.textContent = 'Show Panel';
    }
  });
});
